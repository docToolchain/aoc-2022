func parseInput(input string) [][]int {
	elves := [][]int{}
	elveCounter := 0
	scanner := bufio.NewScanner(strings.NewReader(input))
	for scanner.Scan() {
		line := scanner.Text()
		if line != "" {
			if len(elves) <= elveCounter {
				elves = append(elves, []int{})
			}

		}
	}
